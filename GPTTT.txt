        int request[] = new int[n];

        for(int i = 0; i < n; i++)
        {
            System.out.print("Enter request for process " + i + " (0/1): ");
            request[i] = sc.nextInt();

        System.out.println("\nToken Ring Execution:");

        for(int i = 0; i < n; i++)

            {

            }
            else
            {
                System.out.println("Process " + token + " does not request CS");
            }

            token = (token + 1) % n;
        }

        sc.close();
    }
